-- Bools (i think)
local AimEnabled = false
local AimFOV = 90
local AimSmoothness = 0.15
local InfiniteJumpEnabled = false
UserInputService.JumpRequest:Connect(function()
if InfiniteJumpEnabled then
local Humanoid = Player.Character and Player.Character:FindFirstChildOfClass("Humanoid")
if Humanoid then
Humanoid:ChangeState("Jumping")
end
end
end)
if AimEnabled then
    RunService.RenderStepped:Connect(function()
    if not AimEnabled then return end

    local target = getClosestEnemy()
    if target then
        local cf = CFrame.new(Camera.CFrame.Position, target.Position)
        Camera.CFrame = Camera.CFrame:Lerp(cf, AimSmoothness)
    end
end)


local function getClosestEnemy()
    local closest, dist = nil, AimFOV

    for _, plr in pairs(Players:GetPlayers()) do
        if plr ~= Player and plr.Team ~= Player.Team then
            local char = plr.Character
            local head = char and char:FindFirstChild("Head")
            local hum = char and char:FindFirstChildOfClass("Humanoid")

            if head and hum and hum.Health > 0 then
                local pos, onScreen = Camera:WorldToViewportPoint(head.Position)
                if onScreen then
                    local mousePos = UserInputService:GetMouseLocation()
                    local magnitude = (Vector2.new(pos.X, pos.Y) - mousePos).Magnitude
                    if magnitude < dist then
                        dist = magnitude
                        closest = head
                    end
                end
            end
        end
    end

    return closest
end

local function createToggle(parent, text, initialState, callback)
local Toggle = Instance.new("TextButton")
Toggle.Size = UDim2.new(1, -20, 0, 35)

Toggle.Font = Enum.Font.Gotham
Toggle.TextSize = 14
Toggle.TextColor3 = Color3.fromRGB(230, 230, 230)
Toggle.BackgroundColor3 = Color3.fromRGB(35, 35, 35)
Toggle.BorderSizePixel = 0
Toggle.Parent = parent
Instance.new("UICorner", Toggle).CornerRadius = UDim.new(0, 6)
local state = initialState
Toggle.MouseButton1Click:Connect(function()
state = not state
Toggle.Text = text .. ": " .. (state and "ON" or "OFF")
if callback then callback(state) end
end)
end

-- Minimize: move sidebar and pages off-screen
local function moveContentOffscreen(off)
local xOffset = off and 2000 or 0
TweenService:Create(Sidebar, TweenInfo.new(0.3), {Position = UDim2.new(0, 0 + xOffset, 0, topBarHeight)}):Play()
for _, page in pairs(Pages:GetChildren()) do
TweenService:Create(page, TweenInfo.new(0.3), {Position = UDim2.new(0, 130 + xOffset, 0, 50)}):Play()
end
end

Minimize.MouseButton1Click:Connect(function()
if minimized then
TweenService:Create(Main, TweenInfo.new(0.3), {Size = fullSize}):Play()
moveContentOffscreen(false)
else
TweenService:Create(Main, TweenInfo.new(0.3), {Size = UDim2.new(fullSize.X.Scale, fullSize.X.Offset, 0, topBarHeight)}):Play()
moveContentOffscreen(true)
end
minimized = not minimized
end)
